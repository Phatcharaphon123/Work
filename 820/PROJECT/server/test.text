app.post('/users', async (req, res) => {
  try {
    let user = req.body

    const errors = validateData(user)
    if (errors.length > 0) {
      throw { 
        message: 'กรอกข้อมูลไม่ครบ',
        errors: errors 
      }
    }

    // แก้ SQL query เพื่อใช้คำสั่ง INSERT INTO ตาราง educational
    const results = await conn.query('INSERT INTO educational SET ?', user)
    res.json({
      message: 'insert ok',
      data: results[0]
    })
  } catch (error) {
    const errorMessage = error.message || 'something wrong'
    const errors = error.errors || []
    console.error('error message', error.message)
    res.status(500).json({
      message: errorMessage,
      errors: errors
    })
  }
})
